RGB = 0...255
    0 = ausencia
    255 = maximo
ex: azul = rgb(0,0,255)
    branca = rgb(255,255,255)

# r_ r_ g_ g_ b_ b_
r = Red
g = Green
b = Blue

0 = ausencia de cor

F = maximo de cor

ex: 
    cor preta = #000000
    cor branca = #FFFFFF
    cor vermelha = FF0000
